```baa
// دالة لحساب العامل (Factorial)
عدد_صحيح عامل(عدد_صحيح ن) {
    إذا (ن <= ١) {
        إرجع ١.
    }
    إرجع ن * عامل(ن - ١).
}

// دالة لحساب أعداد فيبوناتشي
عدد_صحيح فيبوناتشي(عدد_صحيح ن) {
    إذا (ن <= ١) {
        إرجع ن.
    }
    إرجع فيبوناتشي(ن - ١) + فيبوناتشي(ن - ٢).
}

// دالة للتحقق من العدد الأولي
منطقي هل_أولي(عدد_صحيح العدد) {
    إذا (العدد < ٢) إرجع خطأ.
    إذا (العدد == ٢) إرجع صحيح.
    إذا (العدد % ٢ == ٠) إرجع خطأ.
    
    لكل (عدد_صحيح ي = ٣؛ ي * ي <= العدد؛ ي += ٢) {
        إذا (العدد % ي == ٠) {
            إرجع خطأ.
        }
    }
    إرجع صحيح.
}

عدد_صحيح رئيسية() {
    // اختبار دالة العامل
    عدد_صحيح ن = ٥.
    اطبع("عامل ").

    اطبع(" = ").
    اطبع_عدد(عامل(ن)).
    
    // اختبار فيبوناتشي
    اطبع("أول ١٠ أرقام فيبوناتشي:");
    لكل (عدد_صحيح ي = ٠؛ ي < ١٠؛ ي++) {
```

### حل معادلة الدرجة الثانية

```baa
#تضمين <math>

// دالة لحل معادلة الدرجة الثانية: ax² + bx + c = 0
فراغ حل_معادلة_ثانية(عدد_حقيقي أ، عدد_حقيقي ب، عدد_حقيقي ج) {
    إذا (أ == ٠) {
        اطبع("هذه ليست معادلة درجة ثانية!");
        إرجع.
    }
    
    عدد_حقيقي المميز = ب * ب - ٤ * أ * ج.
    
    اطبع("المعادلة: ");
    اطبع_عدد_حقيقي(أ);
    اطبع("x² + ");
    اطبع_عدد_حقيقي(ب);
    اطبع("x + ");
    اطبع_عدد_حقيقي(ج);
    اطبع(" = 0");
    
    إذا (المميز > ٠) {
        عدد_حقيقي الجذر١ = (-ب + جذر(المميز)) / (٢ * أ).
        عدد_حقيقي الجذر٢ = (-ب - جذر(المميز)) / (٢ * أ).
        
        اطبع("الجذران حقيقيان ومختلفان:");
        اطبع("الجذر الأول = ");
        اطبع_عدد_حقيقي(الجذر١);
        اطبع("الجذر الثاني = ");
        اطبع_عدد_حقيقي(الجذر٢);
        
    } وإلا إذا (المميز == ٠) {
        عدد_حقيقي الجذر = -ب / (٢ * أ).
        اطبع("الجذر مضاعف:");
        اطبع("الجذر = ");
        اطبع_عدد_حقيقي(الجذر);
        
    } وإلا {
        عدد_حقيقي الجزء_الحقيقي = -ب / (٢ * أ).
        عدد_حقيقي الجزء_التخيلي = جذر(-المميز) / (٢ * أ).
        
        اطبع("الجذران مركبان:");
        اطبع("الجذر الأول = ");
        اطبع_عدد_حقيقي(الجزء_الحقيقي);
        اطبع(" + ");
        اطبع_عدد_حقيقي(الجزء_التخيلي);
        اطبع("i");
        
        اطبع("الجذر الثاني = ");
        اطبع_عدد_حقيقي(الجزء_الحقيقي);
        اطبع(" - ");
        اطبع_عدد_حقيقي(الجزء_التخيلي);
        اطبع("i");
    }
}

عدد_صحيح رئيسية() {
    عدد_حقيقي أ، ب، ج.
    
    اطبع("=== حل معادلة الدرجة الثانية ===");
    اطبع("أدخل معاملات المعادلة ax² + bx + c = 0");
    
    اطبع("أدخل قيمة a: ");
    اقرأ_عدد_حقيقي(&أ);
    اطبع("أدخل قيمة b: ");
    اقرأ_عدد_حقيقي(&ب);
    اطبع("أدخل قيمة c: ");
    اقرأ_عدد_حقيقي(&ج);
    
    حل_معادلة_ثانية(أ، ب، ج);
```

```baa
// دالة لإضافة طالب جديد
فراغ أضف_طالب() {

        الطالب_الجديد->درجة_الرياضيات،
```

```baa
فراغ ترتيب_فقاعي(عدد_صحيح المصفوفة[]، عدد_صحيح الحجم) {

        لكل (عدد_صحيح ك = ٠؛ ك < الحجم - ي - ١؛ ك++) {
            إذا (المصفوفة[ك] > المصفوفة[ك + ١]) {

                عدد_صحيح مؤقت = المصفوفة[ك].
                المصفوفة[ك] = المصفوفة[ك + ١].

        }
    }
```

```baa
// دالة البحث الثنائي
عدد_صحيح بحث_ثنائي(عدد_صحيح المصفوفة[]، عدد_صحيح الحجم، عدد_صحيح الهدف) {
    عدد_صحيح اليسار = ٠.
    عدد_صحيح اليمين = الحجم - ١.
    
    طالما (اليسار <= اليمين) {
        عدد_صحيح الوسط = اليسار + (اليمين - اليسار) / ٢.
        
        إذا (المصفوفة[الوسط] == الهدف) {
            إرجع الوسط.  // وُجد العنصر
        }
        
        إذا (المصفوفة[الوسط] < الهدف) {
            اليسار = الوسط + ١.
        } وإلا {
            اليمين = الوسط - ١.
        }
    }
    
    إرجع -١.  // لم يوجد العنصر
}
```
